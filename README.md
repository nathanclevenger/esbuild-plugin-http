# ESBuild Plugin to allow HTTP Imports

```
const esbuild =  require('esbuild')
const { httpPlugin } = require('../index.js')

esbuild.build({
    entryPoints: ['./test/worker.js'],
    bundle: true,
    outfile: './test/dist.js',
    plugins: [httpPlugin],
  }).catch(() => process.exit(1))
```

worker.js:
```
import { apis, json } from 'https://pkg.do/apis.do@0.1.0'

export default {
    fetch: () => json({apis})
}
```
